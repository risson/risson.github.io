Below is the R code for generating the correct student number attributed to a particular lead submitted date.

The steps are:

  1. Input file is the detail leads information from VS3, with 2 columns "Lead.Submitted.Date" and "Odin.First.Paid.Date".
  2. Create a new column called "StudentOfTheDate". If "Odin.First.Paid.Date" is later than "Lead.Submitted.Date", then mark "StudentOfTheDate" as 1.
  3. Summarize "StudentOfTheDate" by "Lead.Submitted.Date", and generate a table of 2 columns: "DayIndex" and "StudentOfTheDate" - table A
  4. Read the original file provided by Julia: "data for Jeremy report- by day.xlsx" (changed it into csv) - table B
  5. Merge the 2 files together, with a new column called StudentOfTheDate. (B left join A)
  6. Make 2 scatterplots of the old and new calculation of students by date
  7. Write the new file as csv.
library(plyr)
library(dplyr)
library(data.table)

# A function to calculate number of leads become students attributed to the lead submitted date
#   Input: the url of source data about detail leads from VS3
#   Output: a data table of 2 columns, with student number correction
CalculateStudentOfTheDate <- function(url) {

# Read csv file.  File format should be:
#    ODIN.Id: Odin ID. 7-digit string.
#    Date.Submitted...: Lead submitted date. Format as "FullMonth dd, YYYY" ("%B %d, %Y")
#    Odin.First.Paid.Date...: First payment date. Format as "FullMonth dd, YYYY" ("%B %d, %Y")   
DT <- read.csv(file = url)

# Transform into data.table
DT <- data.table(DT)

# Rename columns
DT <- dplyr::rename(DT, DayIndex = Date.Submitted...)
DT <- dplyr::rename(DT, PaidDate = Odin.First.Paid.Date...)

# Transform data class, and mark NAs
DT$DayIndex <- as.character(DT$DayIndex)
DT$PaidDate <- as.character(DT$PaidDate)
DT[DayIndex == "", DayIndex := NA]
DT[PaidDate == "", PaidDate := NA]

# Transform date format
DT[, DayIndex := as.Date(DayIndex, format = "%B %d, %Y")]
DT[, PaidDate := as.Date(PaidDate, format = "%B %d, %Y")]

# Create new column to mark if the student is from this month
# If PaidDate > DayIndex, then this student attributes to DayIndex
DT[!is.na(PaidDate) & PaidDate > DayIndex, StudentOfTheDate := 1]
DT[is.na(StudentOfTheDate), StudentOfTheDate := 0]

# Output is a table with 2 columns:
#   DayIndex: The date customer fill out the lead form
#   StudentOfTheDate: Number of leads become student on that date
NewStudentOfTheDate <- DT %>% dplyr::group_by(DayIndex) %>% dplyr::summarize(StudentOfTheDate = sum(StudentOfTheDate))
}


# A function to allocate refined numbers of students attributed to the lead submitted date
#   Input: 
#       InputUrl: the url of source data provided by Julia, a summary by day
#       OutputUrl:  the url of output data, as csv
#       NewStudentOfTheDate: The data table of 2 columns, with student number correction, the output of function CalculateStudentOfTheDate
#   Output: write a new csv file with refined numbers of students attributed to the lead submitted date
AllocateStudent <- function(InputUrl, OutputUrl, NewStudentOfTheDate) {

    # Read csv file.  File format should be:
    #    DayIndex: Date formated as "m/d/yyyy" ("%m/%d/%Y")
    #    Pageviews: number of pageviews on a particular date
    #    BounceRate: bounce rate on a particular date
    #    PagesPerSession: average pages per session on a particular date
    #    uniquePageviews: unique page views on a particular date
    #    Leads: total number of leads generated on a particular date
    #    Student: total number of student generated on a particular date
    DT <- read.csv(file = InputUrl)

    # Transform into data.table
    DT <- data.table(DT)    

    # Transform date format
    DT$DayIndex <- as.Date(DT$DayIndex, "%m/%d/%Y")

    # set the ON clause as keys of the tables:
    setkey(DT,DayIndex)
    setkey(NewStudentOfTheDate,DayIndex)

    # left join
    Result <- NewStudentOfTheDate[DT]

    # Reset NA
    Result[is.na(StudentOfTheDate), StudentOfTheDate := 0]

    # Show histogram of each
    par(mfcol=c(2,1), mar=c(2,2,2,2))
    plot(Result$DayIndex, Result$Student)
    plot(Result$DayIndex, Result$StudentOfTheDate)

    # Write the csv file with refined numbers of students attributed to the lead submitted date
    write.csv(Result, file = OutputUrl)
}
